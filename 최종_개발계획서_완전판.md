```typescript
// lib/chat/detect.ts (또는 기존 파일)

function detectShowMeIntent(text: string): boolean {
  const patterns = [
    /보여\s*줘/,
    /보여\s*주세요/,
    /사진\s*보여/,
    /이미지\s*보여/,
    /어떻게\s*생겼/,
    /어떤\s*곳/,
  ];
  return patterns.some(p => p.test(text));
}

function extractShowMeQuery(text: string): string {
  // "후쿠오카 맛집 보여줘" → "후쿠오카 맛집"
  return text
    .replace(/보여\s*(줘|주세요|주시오)/g, '')
    .replace(/사진/g, '')
    .replace(/이미지/g, '')
    .trim();
}
```

```typescript
export function googleImageSearch(query: string): string {
  // 구글 이미지 검색 URL
  return `https://www.google.com/search?q=${encodeURIComponent(query)}&tbm=isch`;
```
